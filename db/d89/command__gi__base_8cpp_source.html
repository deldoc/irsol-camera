24std::vector<out_message_t>
25CommandGIBaseHandler::process(
26 std::shared_ptr<irsol::server::ClientSession> session,
27 IRSOL_MAYBE_UNUSED protocol::Command&& message)
28{
29
30 auto& collector = this->ctx.app.frameCollector();
31 auto& state = session->userData().frameListeningState;
32
33 if(state.running()) {
34 IRSOL_NAMED_LOG_WARN(session->id(), "Session already listening to frames. Ignoring request.");
35 std::vector<out_message_t> result;
36 result.emplace_back(protocol::Error::from(message, "Session is already listening to frames"));
37 return result;
38 }
39
40 if(auto errors = validate(message, session); errors.size()) {
41 return errors;
42 }
43
44 auto queue = collector.makeQueuePtr();
45 startListeningThread(session, queue, std::move(message), getDescription(message, session));
46
47 const uint64_t numFrames = getInputSequenceLength(message, session);
48 const double fps = getFrameRate(message, session);
49 collector.registerClient(session->id(), fps, queue, numFrames);
50
51 IRSOL_NAMED_LOG_INFO(session->id(), "Client registered for {} frames at FPS {}", numFrames, fps);
52 return {};
53}

76 std::shared_ptr<std::atomic<bool>> stopRequest) mutable {
77 // Reset the state of the user-data related to frame-listening
78 auto& state = session->userData().frameListeningState;
79 state.gisParams.inputSequenceNumber = 0;
80
81 IRSOL_NAMED_LOG_INFO(
82 session->id(), "Started frame listening thread for {}", message.toString());
83
84 std::unique_ptr<frame_collector::Frame> framePtr;
85 bool interrupted = false;
86 while(!interrupted && (!queue->done() && queue->pop(framePtr))) {
87 IRSOL_NAMED_LOG_DEBUG(
88 session->id(),
89 "Sending frame {} to client: {}",
90 state.gisParams.inputSequenceNumber,
91 framePtr->image.toString());
92 {
93 auto lock = std::scoped_lock(session->socketMutex());
94 std::vector<irsol::protocol::OutMessage> result;
95 result.emplace_back(irsol::protocol::OutMessage(std::move(framePtr->image)));
96 result.emplace_back(irsol::protocol::Success::asStatus(
97 "isn", {static_cast<int>(state.gisParams.inputSequenceNumber)}));
98 session->handleOutMessages(std::move(result));
99 }
100 ++state.gisParams.inputSequenceNumber;
101 if(stopRequest->load()) {
102 interrupted = true;
103 }
104 }
105
106 if(interrupted) {
107 IRSOL_NAMED_LOG_INFO(
108 session->id(), "Stopping execution of frame-collection due to stop-request.");
109 // TODO: on user-stop requests do we want to return a success?
110 return;
111 }
112
113 {
114 auto lock = std::scoped_lock(session->socketMutex());
115 session->handleOutMessage(protocol::Success::from(message));
116 }
117 IRSOL_NAMED_LOG_INFO(session->id(), "{} frame sending complete", message.toString());
118 },
119 description);
120}
