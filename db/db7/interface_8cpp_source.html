171irsol::types::duration_t
172Interface::setExposure(irsol::types::duration_t exposure)
173{
174 IRSOL_ASSERT_ERROR(exposure.count() > 0, "Cannot set non-positive exposure");

176 std::chrono::duration_cast<std::chrono::microseconds>(exposure).count();
177 auto setExposureInMicroseconds =
178 setParam("ExposureTime", static_cast<int64_t>(exposureInMicroseconds));

197void
198Interface::setMultiParam(const std::unordered_map<std::string, camera_param_t>& params)
199{
200 IRSOL_LOG_DEBUG("Setting multiple parameters");
201 std::scoped_lock<std::mutex> lock(m_camMutex);
202 for(const auto& [param, value] : params) {
203 std::visit(
204 [&param, this](auto&& arg) {
205 using U = std::decay_t<decltype(arg)>;
206 IRSOL_LOG_TRACE("Setting parameter '{}' to value '{}'", param, arg);
207 setParamNonThreadSafe<U>(param, arg);
208 },
209 value);
210 }

226Interface::image_t
227Interface::captureImage(std::optional<irsol::types::duration_t> timeout)
228{
229 std::scoped_lock<std::mutex> lock(m_camMutex);
230
231 // Send software trigger to get an image
232 trigger("AcquisitionStart");
233 trigger("TriggerSoftware");
234 // Wait for image, either using the current cached exposure time with a small buffer
235 // or using the user-provided timeout
236 irsol::types::duration_t actualTimeout = m_CachedExposureTime + std::chrono::milliseconds(60);
237 if(timeout.has_value()) {
238 IRSOL_LOG_DEBUG(
239 "User provided a custom timeout of capturing camera of {}",
240 irsol::utils::durationToString(*timeout));
241 actualTimeout = *timeout;
242 } else {
243 IRSOL_LOG_DEBUG(
244 "Using exposure from camera {} with buffer", irsol::utils::durationToString(actualTimeout));
245 }
246
247 uint32_t timeoutMs = static_cast<uint32_t>(
248 std::chrono::duration_cast<std::chrono::milliseconds>(actualTimeout).count());
249 auto image = m_cam.GetImage(timeoutMs);
250 if(image.IsEmpty() || image.GetSize() == 0) {
251 IRSOL_LOG_WARN("Timeout or empty image received.");
252 }
253 trigger("AcquisitionStop");
254 return image;
255}
