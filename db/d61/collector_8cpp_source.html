63void
64FrameCollector::registerClient(
65 irsol::types::client_id_t clientId,
66 double fps,
67 std::shared_ptr<FrameCollector::frame_queue_t> queue,
68 int64_t frameCount)
69{
70
71 std::scoped_lock<std::mutex> lock(m_clientsMutex);
72
73 std::chrono::microseconds interval;
74 bool immediate = false;
75 if(frameCount == 1 && fps <= 0.0) {
76 // Client is requesting a single frame immediately
77 interval = std::chrono::microseconds(1);
78 immediate = true;
79 frameCount = 1;
80 fps = 0.0;
81 } else {
82 interval = std::chrono::microseconds(static_cast<uint64_t>(1000000.0 / fps));
83 IRSOL_NAMED_LOG_INFO(
84 "frame_collector",
85 "registering client with interval of {}",
86 irsol::utils::durationToString(interval));
87 }
88
89 // Registers the client so that the next due time is in SLACK ms.
90 // This is to allow the collector thread to batch multiple clients, that desire a frame at a
91 // specific timestamp, but all within the SLACK duration and to serve them all with the same frame
92 // image.
93 auto nextDue = irsol::types::clock_t::now() + FrameCollector::SLACK;
94
95 IRSOL_NAMED_LOG_INFO(
96 "frame_collector",
97 "Registering client {} with frame rate {} fps (interval {}), "
98 "next frame due at {} #frames {}, immediate {}",
99 clientId,
100 fps,
101 irsol::utils::durationToString(interval),
102 irsol::utils::timestampToString(nextDue),
103 frameCount,
104 immediate);
105
106 m_clients.emplace(
107 clientId, ClientCollectionParams(fps, interval, nextDue, queue, frameCount, immediate));
108 schedule(clientId, nextDue);
109}

118void
119FrameCollector::run()
120{
121 std::unique_lock<std::mutex> lock(m_clientsMutex);
122
123 while(!m_stop) {
124 if(m_scheduleMap.empty()) {
125 // Wait until at least one new client is registered in the map, or if a stop request has
126 // arrived.
127 m_scheduleCondition.wait(lock, [this]() {
128 IRSOL_NAMED_LOG_DEBUG(
129 "frame_collector",
130 "Waiting until a client is scheduled (clients size: {}, schedule size: {})",
131 m_clients.size(),
132 m_scheduleMap.size());
133 return m_stop || !m_scheduleMap.empty();
134 });
135 }
136
137 if(m_stop.load()) {
138 IRSOL_NAMED_LOG_INFO("frame_collector", "Requested frame-collection stop");
139 break;
140 }
141
142 // Retrieve the nextDue time from the schedule map.
143 // This map is always sorted from small to high, as it's an ordered container.
144 irsol::types::timepoint_t nextDue = m_scheduleMap.begin()->first;
145 IRSOL_NAMED_LOG_DEBUG(
146 "frame_collector",
147 "Running for frame collection due at {}",
148 irsol::utils::timestampToString(nextDue));
149
150 // Wait at most until the `nextDue` time.
151 // Allow for early break in case of:
152 // - m_stop is set to true due to a stop-request
153 // - a new client is registered with a nextDue time that is smaller than the current nextDue
154 // time
155 m_scheduleCondition.wait_until(lock, nextDue, [this, currentNextDue = nextDue]() {
156 if(m_stop.load()) {
157 // stopped externally, exit the wait
158 return true;
159 }
160 if(m_scheduleMap.empty()) {
161 // Don't wake up early unnecessarily, as there's no schedules in the map
162 return false;
163 }
164
165 // Check if a new schedule has been inserted into the map, which happens
166 // earlier than the due time we captured prior to sleeping.
167 auto newNextDue = m_scheduleMap.begin()->first;
168
169 IRSOL_NAMED_LOG_DEBUG(
170 "frame_collector",
171 "Current first timestamp in the map: {}, current nextDue: {}",
172 irsol::utils::timestampToString(newNextDue),
173 irsol::utils::timestampToString(currentNextDue));
174 return newNextDue < currentNextDue;
175 });
176
177 if(m_stop.load()) {
178 IRSOL_NAMED_LOG_INFO(
179 "frame_collector", "Frame collection stop request received, breaking loop");
180 return;
181 }
182
183 // Refresh the nextDue, as the above condition might have finished due to a new client being
184 // registered earlier than the 'nextDue' time that was initially selected.
185 nextDue = m_scheduleMap.begin()->first;
186
187 // Clients due now or earlier
188 irsol::types::timepoint_t now = irsol::types::clock_t::now();
189 IRSOL_NAMED_LOG_DEBUG(
190 "frame_collector",
191 "Woken up at timestamp {}, with next due at {}",
192 irsol::utils::timestampToString(now),
193 irsol::utils::timestampToString(nextDue));
194
195 const auto slack = m_clients.size() == 1 ? std::chrono::milliseconds(0) : FrameCollector::SLACK;
196 auto [readyClients, clientsSchedules] = collectReadyClients(now, slack);
197 IRSOL_NAMED_LOG_DEBUG(
198 "frame_collector", "Found {} clients that need an image now!", readyClients.size());
199 {
200 auto uniqueSchedules =
201 std::set<irsol::types::timepoint_t>(clientsSchedules.begin(), clientsSchedules.end());
202 for(auto uniqueSchedule : uniqueSchedules) {
203 auto numClientsWithSchedule = std::count_if(
204 clientsSchedules.begin(),
205 clientsSchedules.end(),
206 [&uniqueSchedule](const auto& schedule) { return schedule == uniqueSchedule; });
207
208 if(uniqueSchedule != nextDue) {
209 IRSOL_NAMED_LOG_WARN(
210 "frame_collector_slack",
211 "Actual schedule: {}, also considering {} clients with schedule {} due to allowed "
212 "slack of {}",
213 irsol::utils::timestampToString(nextDue),
214 numClientsWithSchedule,
215 irsol::utils::timestampToString(uniqueSchedule),
216 irsol::utils::durationToString(slack));
217 }
218 }
219 }
220 cleanUpSchedule(clientsSchedules);
221 IRSOL_NAMED_LOG_DEBUG(
222 "frame_collector",
223 "After schedule cleanup, there are still {} schedules",
224 m_scheduleMap.size());
225
226 auto [frameMetadata, imageRawBuffer] = grabImageData();
227
228 // Deliver the frame to clients
229 std::vector<irsol::types::client_id_t> finishedClient;
230 for(auto clientId : readyClients) {
231 IRSOL_NAMED_LOG_DEBUG("frame_collector", "Notifying client {} for new image data", clientId);
232 auto& clientParams = m_clients.at(clientId);
233 // Push a new frame created on the fly to the current client's queue.
234 // This creates a copy of the image data into the queue, and this is wanted, so that if a
235 // consumer modifies the image, it doesn't affect other clients.
236 clientParams.queue->push(std::make_unique<Frame>(
237 frameMetadata,
238 irsol::protocol::ImageBinaryData(
239 {imageRawBuffer.begin(), imageRawBuffer.end()},
240 {frameMetadata.height, frameMetadata.width},
241 {irsol::protocol::BinaryDataAttribute("imageId", static_cast<int>(frameMetadata.frameId)),
242 irsol::protocol::BinaryDataAttribute(
243 "timestamp", irsol::utils::timestampToString(frameMetadata.timestamp))})));
244
245 // Try to schedule the client, if no longer needed, register it in the finishedClients
246 if(!schedule(clientId, clientParams.nextFrameDue + clientParams.interval)) {
247 finishedClient.push_back(clientId);
248 }
249 }
250
251 // Unlock the clients so we can remove finished clients
252 for(const auto& clientId : finishedClient) {
253 IRSOL_NAMED_LOG_DEBUG(
254 "frame_collector",
255 "Deregistering client {}, as it has consumed all the frames it needed.",
256 clientId);
257 deregisterClientNonThreadSafe(clientId);
258 }
259
260 IRSOL_NAMED_LOG_DEBUG("frame_collector", "Loop finished, restarting loop");
261 }

264void
265FrameCollector::deregisterClientNonThreadSafe(irsol::types::client_id_t clientId)
266{
267 // Mark the client's queue as finished, so the client is notified that
268 // no more data will be pushed to him
269 IRSOL_NAMED_LOG_INFO("frame_collector", "Deregistering client {}", clientId);
270
271 if(m_clients.find(clientId) == m_clients.end()) {
272 IRSOL_NAMED_LOG_WARN(
273 "frame_collector", "Client {} was already deregistered, ignoring request.", clientId);
274 return;
275 }
276 auto& clientParams = m_clients.at(clientId);
277 clientParams.queue->producerFinished();
278
279 // Removes the client from the storage
280 auto dueNext = clientParams.nextFrameDue;
281
282 m_clients.erase(clientId);
283 IRSOL_NAMED_LOG_DEBUG(
284 "frame_collector", "Removed client {}, now remaining {} clients", clientId, m_clients.size());
285
286 if(m_scheduleMap.find(dueNext) == m_scheduleMap.end()) {
287 IRSOL_NAMED_LOG_WARN(
288 "frame_collector",
289 "Scheduled time {} was already deregistered. Ignoring request.",
290 irsol::utils::timestampToString(dueNext));
291 return;
292 }
293 auto& clientsAtDue = m_scheduleMap.at(dueNext);
294
295 IRSOL_NAMED_LOG_DEBUG(
296 "frame_collector",
297 "Deregistering client {} from next schedule at {}",
298 clientId,
299 irsol::utils::timestampToString(dueNext));
300 clientsAtDue.erase(
301 std::remove(clientsAtDue.begin(), clientsAtDue.end(), clientId), clientsAtDue.end());
302
303 if(clientsAtDue.empty()) {
304 IRSOL_NAMED_LOG_DEBUG(
305 "frame_collector",
306 "No more clients for schedule {}. Removing schedule.",
307 irsol::utils::timestampToString(dueNext));
308 m_scheduleMap.erase(dueNext);
309 }
310
311 m_scheduleCondition.notify_one();
312}

314std::pair<std::vector<irsol::types::client_id_t>, std::vector<irsol::types::timepoint_t>>
315FrameCollector::collectReadyClients(irsol::types::timepoint_t now, irsol::types::duration_t slack)
316 const
317{
318 IRSOL_NAMED_LOG_DEBUG(
319 "frame_collector",
320 "Collecting clients for time {}, with slack of {}",
321 irsol::utils::timestampToString(now),
322 irsol::utils::durationToString(slack));
323 std::vector<irsol::types::client_id_t> readyClients;
324 std::vector<irsol::types::timepoint_t> clientsSchedule;
325
326 for(const auto& [scheduleTime, clients] : m_scheduleMap) {
327 // As soon as the scheduleTime of the client is above now+slack, break the loop.
328 // We can break early, as m_scheduleMap is an ordered container, that is iterated over
329 // in a way that smaller keys (scheduledTimes) are always iterated over at the beginning.
330 if(scheduleTime > now + slack) {
331 break;
332 }
333 for(const auto& client : clients) {
334 readyClients.push_back(client);
335 clientsSchedule.push_back(scheduleTime);
336 }
337 }
338
339 return {readyClients, clientsSchedule};
340}

351std::pair<FrameMetadata, std::vector<irsol::types::byte_t>>
352FrameCollector::grabImageData() const
353{
354 // Capture the frame just-in-time
355 IRSOL_MAYBE_UNUSED auto t0 = irsol::types::clock_t::now();
356 auto image = m_cam.captureImage();
357 IRSOL_MAYBE_UNUSED auto t1 = irsol::types::clock_t::now();
358 IRSOL_NAMED_LOG_DEBUG(
359 "frame_collector",
360 "Capture image: start: {}, stop: {}, duration: {}",
361 irsol::utils::timestampToString(t0),
362 irsol::utils::timestampToString(t1),
363 irsol::utils::durationToString(t1 - t0));
364
365 // Extract the image data from the image buffer, and copy it into an
366 // owning structure. In this way, when `image` is destroyed at the end of the
367 // execution of this function, it can return into the pool of NeoAPI::Images
368 // for next frames to be written to the buffer.
369 auto* imageData = image.GetImageData();
370 auto numBytes = image.GetSize();
371
372 std::vector<irsol::types::byte_t> rawData(numBytes);
373 std::memcpy(rawData.data(), imageData, numBytes);
374
375 return std::make_pair<FrameMetadata, std::vector<irsol::types::byte_t>>(
376 {irsol::types::clock_t::now(), image.GetImageID(), image.GetHeight(), image.GetWidth()},
377 std::move(rawData));
378}

380bool
381FrameCollector::schedule(
382 const irsol::types::client_id_t& clientId,
383 irsol::types::timepoint_t nextFrameDue)
384{
385 IRSOL_ASSERT_ERROR(
386 m_clients.find(clientId) != m_clients.end(), "Impossible to schedule an unregistered client.");
387 // Update the parameters of the client.
388 auto& clientParams = m_clients.at(clientId);
389 if(clientParams.remainingFrames-- == 0 && clientParams.remainingFrames < 0) {
390 // Client no longer expects frames.
391 // This handles also clients that are listening forever, as their 'remainingFrames' is negative
392 // so this¨ condition is never fully met.
393 IRSOL_NAMED_LOG_DEBUG(
394 "frame_collector", "Client {} had no longer frames to produce.", clientId);
395 return false;
396 }
397
398 if(clientParams.nextFrameDue == nextFrameDue) {
399 // This has been called the first time for client-registration
400 IRSOL_NAMED_LOG_DEBUG(
401 "frame_collector",
402 "Client {} has been scheduled for timestamp {}.",
403 clientId,
404 irsol::utils::timestampToString(clientParams.nextFrameDue));
405 } else {
406
407 IRSOL_NAMED_LOG_DEBUG(
408 "frame_collector",
409 "(Rescheduling client {} for next frame, previous due: {}, next due {}, # count {}",
410 clientId,
411 irsol::utils::timestampToString(clientParams.nextFrameDue),
412 irsol::utils::timestampToString(nextFrameDue),
413 clientParams.remainingFrames);
414 }
415
416 // Updates the parameters of the client
417 clientParams.nextFrameDue = nextFrameDue;
418
419 // Registers the client for the scheduled timestamp.
420 m_scheduleMap[nextFrameDue].push_back(clientId);
421
422 // Notify the condition variable, that a new client has been scheduled
423 m_scheduleCondition.notify_one();
424 return true;
425}
