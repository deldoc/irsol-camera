17void
18demoCameraFeatures(irsol::camera::Interface& cam)
19{
20 IRSOL_LOG_INFO("=== Original Camera Information ===");
21 IRSOL_LOG_INFO("\n{}", cam.cameraStatusAsString());
22
23 IRSOL_LOG_INFO("=== Reading Common Camera Parameters ===");
24 auto mode = cam.getParam<std::string>("AcquisitionMode");
25 IRSOL_LOG_INFO("AcquisitionMode: {}", mode);
26
27 auto fpsEnable = cam.getParam<bool>("AcquisitionFrameRateEnable");
28 auto fps = cam.getParam<double>("AcquisitionFrameRate");
29 IRSOL_LOG_INFO("AcquisitionFrameRateEnable: {}", fpsEnable);
30 IRSOL_LOG_INFO("AcquisitionFrameRate: {:.2f}", fps);
31
32 auto exposure = cam.getExposure();
33 IRSOL_LOG_INFO("Current exposure time: {}", irsol::utils::durationToString(exposure));
34
35 IRSOL_LOG_INFO("=== Modifying Parameters ===");
36 // Set exposure
37 auto newExposure = std::chrono::microseconds(15000);
38 auto actualNewExposure = cam.setExposure(newExposure);
39 IRSOL_LOG_INFO(
40 "New exposure set to: {}, actual value now: {}",
41 irsol::utils::durationToString(newExposure),
42 irsol::utils::durationToString(actualNewExposure));
43
44 // Set FPS (requires enabling first)
45 auto newAcquisitionFrameRateEnable = true;
46 auto actualAcquisitionFrameRateEnable =
47 cam.setParam("AcquisitionFrameRateEnable", newAcquisitionFrameRateEnable);
48 IRSOL_LOG_INFO(
49 "New acquisition frame rate enablement set to: {}, actual value now: {}",
50 newAcquisitionFrameRateEnable,
51 actualAcquisitionFrameRateEnable);
52 auto newAcquisitionFrameRate = 12.5;
53 auto actualAcquisitionFrameRate = cam.setParam("AcquisitionFrameRate", newAcquisitionFrameRate);
54 IRSOL_LOG_INFO(
55 "New acquisition frame rate set to: {}, actual value now: {}",
56 newAcquisitionFrameRate,
57 actualAcquisitionFrameRate);
58
59 // Change acquisition mode to SingleFrame
60 cam.setParam("AcquisitionMode", "SingleFrame");
61 IRSOL_LOG_INFO("AcquisitionMode set to SingleFrame");
62
63 // Use setMultiParam for grouped settings
64 cam.setMultiParam({
65 {"OffsetX", 0},
66 {"OffsetY", 0},
67 {"Height", 124},
68 });
69 IRSOL_LOG_INFO("Updated Offset and Height window");
70
71 IRSOL_LOG_INFO("=== Capturing Image ===");
72 auto image = cam.captureImage();
73 if(image.IsEmpty()) {
74 IRSOL_LOG_WARN("Image is empty.");
75 } else {
76 IRSOL_LOG_INFO(
77 "Captured image: {}x{} ({} bytes)", image.GetHeight(), image.GetWidth(), image.GetSize());
78 }
79
80 IRSOL_LOG_INFO("=== Reading Back New Parameter Values ===");
81 IRSOL_LOG_INFO("AcquisitionFrameRate: {:.2f}", cam.getParam<double>("AcquisitionFrameRate"));
82 IRSOL_LOG_INFO("ExposureTime: {} ", irsol::utils::durationToString(cam.getExposure()));
83 IRSOL_LOG_INFO("AcquisitionMode: {}", cam.getParam<std::string>("AcquisitionMode"));
84
85 IRSOL_LOG_INFO("=== Final Camera Information ===");
86 IRSOL_LOG_INFO("\n{}", cam.cameraStatusAsString());
87}

90int
91main(int argc, char** argv)
92{
93 // Initialize logging
94 std::string logPath = "logs/" + getProgramName() + ".log";
95 irsol::initLogging(logPath.c_str(), spdlog::level::debug);
96 irsol::initAssertHandler();
97
98 // Create camera interface
99 auto cam = irsol::camera::Interface::HalfResolution();
100
101 IRSOL_ASSERT_FATAL(cam.isConnected(), "Camera is not connected");
102
103 // Run feature demo
104 demoCameraFeatures(cam);
105
106 return 0;
107}
