142 IRSOL_LOG_ERROR(

144 } else if constexpr(std::is_same_v<T, protocol::Command>) {
145 IRSOL_LOG_ERROR(
146 "Duplicate handler registered for identifier '{}' and Command type", identifier);
147 } else {

149 }
150
151 return false;
152 }

154 if constexpr(std::is_same_v<T, protocol::Assignment>) {
155 IRSOL_LOG_INFO("Registering handler for 'Assignment(\"{}\")'", identifier);
156 m_assignmentMessageHandlers[identifier] = handler;
157 } else if constexpr(std::is_same_v<T, protocol::Inquiry>) {

162 m_commandMessageHandlers[identifier] = handler;
