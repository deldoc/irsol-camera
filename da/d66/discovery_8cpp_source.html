15enum FeaturePermissionValue
16{
17 UNSET = 0,
18 NOT_AVAILABLE = (1u << 0),
19 AVAILABLE = (1u << 1),
20 READABLE = (1u << 2),
21 WRITABLE = (1u << 3)
22};

24struct FeaturePermissions

54std::map<FeaturePermissions, std::vector<NeoAPI::Feature*>>
55extractCameraFeatures(NeoAPI::Cam& cam)
56{
57 std::map<internal::FeaturePermissions, std::vector<NeoAPI::Feature*>> featurePermissionsMap;
58
59 uint64_t featureCount = 0;
60 for(auto& f : cam.GetFeatureList()) {
61 IRSOL_LOG_TRACE("Feature: {0:s}", f.GetName().c_str());
62 auto isAvailable = f.IsAvailable();
63 auto isReadable = f.IsReadable();
64 auto isWritable = f.IsWritable();
65 internal::FeaturePermissions permissions = {isAvailable, isReadable, isWritable};
66 if(featurePermissionsMap.find(permissions) == featurePermissionsMap.cend()) {
67 featurePermissionsMap[permissions] = std::vector<NeoAPI::Feature*>();
68 }
69 featurePermissionsMap[permissions].push_back(&f);
70 featureCount++;
71 }
72
73 IRSOL_SUPPRESS_UNUSED_STRUCTURED_BINDING_START // Disable unused variable warning
74 for(auto& [_, features] : featurePermissionsMap)
75 {
76 IRSOL_SUPPRESS_UNUSED_STRUCTURED_BINDING_END // restores the warning
77 std::sort(features.begin(), features.end(), [](NeoAPI::Feature* a, NeoAPI::Feature* b) {
78 return std::string(a->GetName()) < std::string(b->GetName());
79 });
80 }
81
82 IRSOL_LOG_INFO("Loaded {0:d} camera features", featureCount);
83 return featurePermissionsMap;
84}

90void
91FeatureDiscovery::run()
92{
93 const auto& featurePermissionsMap = internal::extractCameraFeatures(m_cam.getNeoCam());
94 for(const auto& [permissions, features] : featurePermissionsMap) {
95 IRSOL_LOG_INFO(
96 "Permissions: isAvailable: {0}, isReadable: {1}, isWritable: {2}",
97 permissions.isAvailable(),
98 permissions.isReadable(),
99 permissions.isWritable());
100 for(const auto feature : features) {
101 std::string featureDescription{feature->GetName().c_str()};
102 if(permissions.value & internal::FeaturePermissionValue::AVAILABLE) {
103 if(permissions.value & internal::FeaturePermissionValue::READABLE) {
104 featureDescription += ", readable (";
105 featureDescription += NeoAPI::NeoString(*feature).c_str();
106 featureDescription += ")";
107 }
108 if(permissions.value & internal::FeaturePermissionValue::WRITABLE) {
109 featureDescription += ", writable";
110 }
111 } else {
112 featureDescription += ", not available";
113 }
114
115 IRSOL_LOG_INFO(" - {0:s}", featureDescription);
116 }
117 }
