template<typename T, typename VariantT>
struct irsol::traits::is_type_in_variant< T, VariantT >

Checks whether a type T is one of the types in a std::variant.

Evaluates to true if T (after std::decay) matches any of the types contained in the specified variant. Used for enabling template functions or static dispatch based on allowed types.

using VariantType = std::variant<int, float>;
static_assert(irsol::traits::is_type_in_variant<int, VariantType>::value, "int is valid");
static_assert(!irsol::traits::is_type_in_variant<std::string, VariantType>::value, "string is not
valid");
