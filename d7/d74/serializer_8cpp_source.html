34internal::SerializedMessage
35Serializer::serializeSuccess(Success&& msg)
36{
37 IRSOL_LOG_TRACE("Serializing Success message: {}", msg.toString());
38 std::string result = msg.identifier;
39 if(msg.source == InMessageKind::ASSIGNMENT) {
40 IRSOL_ASSERT_ERROR(
41 msg.hasBody(),
42 "Body is not present in 'Success' message, created from 'Assignment'. This should "
43 "never happen, as a successful assignment should always provide a body for the "
44 "associated Success message.");
45 auto body = *msg.body;
46 result += "=" + serializeValue(std::move(body)) + Serializer::message_termination;
47 } else if(msg.source == InMessageKind::INQUIRY) {
48 if(msg.hasBody()) {
49 auto body = *msg.body;
50 result += "=" + serializeValue(std::move(body)) + Serializer::message_termination;
51 } else {
52 result += Serializer::message_termination;
53 }
54 } else if(msg.source == InMessageKind::COMMAND) {
55 result += ";\n";
56 }
57 return {result, {}};
58}
