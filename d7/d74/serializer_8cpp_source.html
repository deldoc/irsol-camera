35internal::SerializedMessage
36Serializer::serializeSuccess(Success&& msg)
37{
38 IRSOL_LOG_TRACE("Serializing Success message: {}", msg.toString());
39 std::string result = msg.identifier;
40 if(msg.source == InMessageKind::ASSIGNMENT) {
41 IRSOL_ASSERT_ERROR(
42 msg.hasBody(),
43 "Body is not present in 'Success' message, created from 'Assignment'. This should "
44 "never happen, as a successful assignment should always provide a body for the "
45 "associated Success message.");
46 auto body = *msg.body;
47 result += "=" + serializeValue(std::move(body)) + Serializer::message_termination;
48 } else if(msg.source == InMessageKind::INQUIRY) {
49 if(msg.hasBody()) {
50 auto body = *msg.body;
51 result += "=" + serializeValue(std::move(body)) + Serializer::message_termination;
52 } else {
53 result += Serializer::message_termination;
54 }
55 } else if(msg.source == InMessageKind::COMMAND) {
56 result += ";\n";
57 }
58 return {result, {}};
59}

69internal::SerializedMessage
70Serializer::serializeImageBinaryData(ImageBinaryData&& msg)
71{
72 IRSOL_LOG_TRACE("Serializing image binary data: {}", msg.toString());
73
74 std::vector<irsol::types::byte_t> payload;
75 payload.reserve(msg.data.size());
76 {
77 std::string imagePrefix = "img=";
78 auto imagePrefixAsBytes = irsol::utils::stringToBytes(imagePrefix);
79 std::move(imagePrefixAsBytes.begin(), imagePrefixAsBytes.end(), std::back_inserter(payload));
80 }
81 payload.emplace_back(Serializer::SpecialBytes::SOH);
82 {
83 std::stringstream ss;
84 ss << "u" << msg.BYTES_PER_ELEMENT * 8 << "[" << msg.shape[0] << "," << msg.shape[1] << "]";
85 for(auto& att : msg.attributes) {
86 ss << " " << serializeBinaryDataAttribute(std::move(att));
87 }
88 auto attributesString = ss.str();
89 IRSOL_LOG_DEBUG(
90 "Attributes for message '{}' serialized to {}", msg.toString(), attributesString);
91 auto attributesStringAsBytes = irsol::utils::stringToBytes(attributesString);
92
93 payload.insert(payload.end(), attributesStringAsBytes.begin(), attributesStringAsBytes.end());
94 }
95 payload.emplace_back(Serializer::SpecialBytes::STX);
96 std::move(msg.data.begin(), msg.data.end(), std::back_inserter(payload));
97 payload.emplace_back(Serializer::SpecialBytes::ETX);
98
99 auto message = internal::SerializedMessage("", std::move(payload));
100 return std::move(message);
101}
